```JS
var postorderTraversal = (root) => {
    let res = []
    let stack = []
    
    if (root)	stack.push(root)
    while (stack.length > 0) {
		let node = stack.pop()
        
        res.unshift(node.val)
        
        if (node.left != null) 
            stack.push(node.left)
        if (node.right != null) 
            stack.push(node.right)
    }
    return res
}
```
